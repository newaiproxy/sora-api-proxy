import http.client
import json

conn = http.client.HTTPSConnection("")
payload = json.dumps({
   "prompt": "make animate",
   "model": "sora-2",
   "aspect_ratio": "16:9",
   "hd": False,
   "duration": "10"
})
headers = {
   'Authorization': 'Bearer {{YOUR_API_KEY}}',
   'Content-Type': 'application/json'
}
conn.request("POST", "/v2/videos/generations", payload, headers)
res = conn.getresponse()
data = res.read()
print(data.decode("utf-8"))
